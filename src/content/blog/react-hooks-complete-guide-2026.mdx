# React Hooks Complete Guide with Practical Examples 2026

React hooks transform stateless functional components into powerful, expressive units. Mastering all hooks — including the newer React 18+ and React 19 additions — is essential for modern React development.

## useState: Managing Component State

```typescript
// Basic state
const [count, setCount] = useState(0);

// Object state — always spread to preserve other fields
const [form, setForm] = useState({ name: '', email: '', active: true });
const updateField = (field: string, value: string) =>
  setForm(prev => ({ ...prev, [field]: value }));

// Lazy initial state (for expensive computations)
const [data, setData] = useState(() => parseExpensiveData(rawData));

// Functional update (always use when new state depends on previous)
setCount(prev => prev + 1);
setItems(prev => [...prev, newItem]);
```

## useEffect: Side Effects and Lifecycle

```typescript
// Data fetching
useEffect(() => {
  const abortController = new AbortController();

  async function fetchData() {
    try {
      const response = await fetch(`/api/users/${userId}`, {
        signal: abortController.signal
      });
      const data = await response.json();
      setUser(data);
    } catch (err) {
      if (err.name !== 'AbortError') setError(err);
    }
  }

  fetchData();

  // Cleanup: abort fetch if component unmounts or userId changes
  return () => abortController.abort();
}, [userId]);  // Re-run when userId changes

// Subscribe and unsubscribe
useEffect(() => {
  const subscription = eventBus.subscribe('update', handleUpdate);
  return () => subscription.unsubscribe();  // Cleanup
}, []);

// Dependency rules:
// - [] : run once on mount
// - [dep1, dep2] : run when dep1 or dep2 changes
// - no array : run after every render (rarely correct)
```

## useContext: Consuming Context

```typescript
// Create context with TypeScript
interface ThemeContextType {
  theme: 'light' | 'dark';
  toggleTheme: () => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

// Type-safe context hook
function useTheme(): ThemeContextType {
  const context = useContext(ThemeContext);
  if (!context) throw new Error('useTheme must be used within ThemeProvider');
  return context;
}

// Provider
function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<'light' | 'dark'>('light');
  const toggleTheme = useCallback(() =>
    setTheme(prev => prev === 'light' ? 'dark' : 'light'), []);

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

// Consumer
function Header() {
  const { theme, toggleTheme } = useTheme();
  return <button onClick={toggleTheme}>Current: {theme}</button>;
}
```

## useReducer: Complex State Machines

```typescript
type CartState = {
  items: CartItem[];
  total: number;
  coupon: string | null;
};

type CartAction =
  | { type: 'ADD_ITEM'; item: CartItem }
  | { type: 'REMOVE_ITEM'; itemId: string }
  | { type: 'APPLY_COUPON'; code: string }
  | { type: 'CLEAR_CART' };

function cartReducer(state: CartState, action: CartAction): CartState {
  switch (action.type) {
    case 'ADD_ITEM': {
      const items = [...state.items, action.item];
      return { ...state, items, total: calculateTotal(items) };
    }
    case 'REMOVE_ITEM': {
      const items = state.items.filter(i => i.id !== action.itemId);
      return { ...state, items, total: calculateTotal(items) };
    }
    case 'APPLY_COUPON':
      return { ...state, coupon: action.code };
    case 'CLEAR_CART':
      return { items: [], total: 0, coupon: null };
    default:
      return state;
  }
}

// Usage
const [cart, dispatch] = useReducer(cartReducer, { items: [], total: 0, coupon: null });

dispatch({ type: 'ADD_ITEM', item: { id: '1', name: 'T-Shirt', price: 29.99 } });
dispatch({ type: 'APPLY_COUPON', code: 'SAVE20' });
```

## useRef: Mutable Values and DOM Access

```typescript
// DOM reference
const inputRef = useRef<HTMLInputElement>(null);
const focusInput = () => inputRef.current?.focus();

// Mutable value (doesn't trigger re-render when changed)
const renderCount = useRef(0);
renderCount.current++;  // No re-render

// Storing previous value
function usePrevious<T>(value: T): T | undefined {
  const ref = useRef<T>();
  useEffect(() => { ref.current = value; }, [value]);
  return ref.current;
}

// Interval that always uses latest callback
function useInterval(callback: () => void, delay: number) {
  const savedCallback = useRef(callback);
  useEffect(() => { savedCallback.current = callback; }, [callback]);
  useEffect(() => {
    const id = setInterval(() => savedCallback.current(), delay);
    return () => clearInterval(id);
  }, [delay]);
}
```

## useId: Stable Unique IDs

```typescript
// For accessibility: link label to input
function FormField({ label }: { label: string }) {
  const id = useId();  // Stable across renders, unique per instance

  return (
    <div>
      <label htmlFor={id}>{label}</label>
      <input id={id} type="text" />
    </div>
  );
}

// Multiple IDs from one hook
function Form() {
  const formId = useId();
  const nameId = `${formId}-name`;
  const emailId = `${formId}-email`;

  return (
    <form>
      <label htmlFor={nameId}>Name</label>
      <input id={nameId} name="name" />
      <label htmlFor={emailId}>Email</label>
      <input id={emailId} name="email" />
    </form>
  );
}
```

## useTransition: Non-Urgent Updates (React 18+)

```typescript
const [isPending, startTransition] = useTransition();

function handleTabChange(tab: string) {
  startTransition(() => {
    // This update is non-urgent — React can interrupt it
    // to handle more important updates (like user input)
    setActiveTab(tab);
    setContent(computeTabContent(tab));  // Expensive computation
  });
}

// isPending shows a loading indicator without blocking input
return (
  <div>
    {tabs.map(tab => (
      <button key={tab} onClick={() => handleTabChange(tab)}>
        {tab}
      </button>
    ))}
    {isPending && <LoadingSpinner />}
    <div style={{ opacity: isPending ? 0.7 : 1 }}>
      {content}
    </div>
  </div>
);
```

function SearchResults({ query }: { query: string }) {

## Custom Hooks: Reusable Logic

```typescript
// Data fetching hook
function useData<T>(url: string) {
  const [data, setData] = useState<T | null>(null);
  const [error, setError] = useState<Error | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);

    fetch(url)
      .then(r => r.json())
      .then(data => { if (!cancelled) { setData(data); setLoading(false); } })
      .catch(err => { if (!cancelled) { setError(err); setLoading(false); } });

    return () => { cancelled = true; };
  }, [url]);

  return { data, error, loading };
}

// Local storage hook
function useLocalStorage<T>(key: string, initialValue: T) {
  const [value, setValue] = useState<T>(() => {
    if (typeof window === 'undefined') return initialValue;
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch { return initialValue; }
  });

  const setStoredValue = useCallback((newValue: T | ((prev: T) => T)) => {
    setValue(prev => {
      const resolved = typeof newValue === 'function'
        ? (newValue as (prev: T) => T)(prev)
        : newValue;
      localStorage.setItem(key, JSON.stringify(resolved));
      return resolved;
    });
  }, [key]);

  return [value, setStoredValue] as const;
}

// Usage
const [theme, setTheme] = useLocalStorage('theme', 'light');
const { data: user, loading } = useData<User>('/api/user');
```

Hooks are React's composition primitive. Master these patterns and you can abstract any stateful logic into reusable, testable units.
